# StockExperimentUS Port Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a new repo `StockExperimentUS` that runs the 4 live Indian momentum models against US (Nasdaq) equities, sourcing data from yfinance instead of Fyers, and report each model's 4-year CAGR. Deploy to the network NUC via docker-compose (not the k8s cluster).

**Architecture:** Copy the existing `StockExperiment` repo into `StockExperimentUS`, strip Fyers/India runtime coupling for the backtest path, and introduce a yfinance bulk data loader that writes into the same `historical_data` Postgres table with `data_source='yfinance'` and plain US tickers. The 4 backtests (`tools/models/<m>/backtest.py`) read that table. Universe lists become `nasdaq100.csv` (real Nasdaq-100, mirrors `nifty100.csv`) and `nasdaq500.csv` (top-500 Nasdaq by market cap, mirrors `nifty500.csv`). Backtests are run inside the app container against the dockerized Postgres.

**Tech Stack:** Python 3.11, pandas, yfinance, SQLAlchemy, PostgreSQL 15, Docker Compose. No Fyers, no broker auth in the backtest path.

## Source → US Transform Contract (applies to all model ports)

Every Indian model backtest carries the same India coupling. The US equivalents:

| India | US |
|---|---|
| `data_source='fyers'` (in SQL WHERE) | `data_source='yfinance'` |
| `f"NSE:{s}-EQ"` (DB symbol form) | `s` (plain US ticker, e.g. `AAPL`) |
| `.replace("NSE:","").replace("-EQ","")` (display) | leave as-is (no-op on plain tickers) |
| `from tools.shared.universes import nifty500_symbols` | `from tools.shared.universes import nasdaq500_symbols` |
| `/app/src/data/symbols/nifty100.csv` | `/app/src/data/symbols/nasdaq100.csv` |
| `/app/src/data/symbols/nifty_smallcap250.csv` | (none — smallcap drop disabled for US) |
| `MAX_PRICE = 3000` (₹, share-count floor for ₹30k live) | `MAX_PRICE = 1e9` (effectively off for $1M backtest; live-sizing knob) |
| `₹` in print/report strings | `$` |
| `DEFAULT_START = date(2023,5,15)` | `date(2022,5,24)` (4 years) |
| `DEFAULT_END = date(2026,5,1x)` | `date(2026,5,24)` (today) |
| `DEFAULT_CAP = 1_000_000.0` | unchanged (nominal $1M) |

```python
# tools/refresh_nasdaq100.py
"""Refresh nasdaq100.csv (real Nasdaq-100 constituents) from Wikipedia.

Usage: python tools/refresh_nasdaq100.py
Writes src/data/symbols/nasdaq100.csv with columns: Symbol,Series
(Series='EQ' kept for parity with the India nifty100.csv loader.)
Mirrors the role of nifty100.csv in the Indian repo.
"""
from __future__ import annotations
import sys
from pathlib import Path
import pandas as pd

OUT = Path(__file__).resolve().parent.parent / "src" / "data" / "symbols" / "nasdaq100.csv"
URL = "https://en.wikipedia.org/wiki/Nasdaq-100"


def _sanitize(t: str) -> str:
    # yfinance uses '-' for class shares (BRK.B -> BRK-B)
    return t.strip().upper().replace(".", "-").replace("/", "-")


def main() -> int:
    OUT.parent.mkdir(parents=True, exist_ok=True)
    tables = pd.read_html(URL)
    # The constituents table has a 'Ticker' or 'Symbol' column
    comp = None
    for t in tables:
        cols = {c.lower() for c in map(str, t.columns)}
        if "ticker" in cols or "symbol" in cols:
            comp = t
            break
    if comp is None:
        print("Could not find constituents table", file=sys.stderr)
        return 1
    col = "Ticker" if "Ticker" in comp.columns else "Symbol"
    syms = sorted({_sanitize(s) for s in comp[col].astype(str) if s and s != "nan"})
    with open(OUT, "w") as f:
        f.write("Symbol,Series\n")
        for s in syms:
            f.write(f"{s},EQ\n")
    print(f"Saved {len(syms)} symbols to {OUT}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
# tools/refresh_nasdaq500.py
"""Refresh nasdaq500.csv = top 500 Nasdaq-listed stocks by market cap.

Mirrors nifty500.csv in the Indian repo (the broad ADV-rankable pool).
Source: nasdaq.com screener API (needs a browser User-Agent).
Usage: python tools/refresh_nasdaq500.py
"""
from __future__ import annotations
import sys
from pathlib import Path
import requests

OUT = Path(__file__).resolve().parent.parent / "src" / "data" / "symbols" / "nasdaq500.csv"
API = ("https://api.nasdaq.com/api/screener/stocks"
       "?tableonly=true&limit=6000&exchange=nasdaq")
HEADERS = {
    "User-Agent": "Mozilla/5.0 (Macintosh; Intel Mac OS X 10_15_7)",
    "Accept": "application/json, text/plain, */*",
    "Accept-Language": "en-US,en;q=0.9",
}


def _mktcap(v: str) -> float:
    try:
        return float(str(v).replace("$", "").replace(",", "") or 0)
    except ValueError:
        return 0.0


def _sanitize(t: str) -> str:
    return t.strip().upper().replace(".", "-").replace("/", "-")


def main() -> int:
    OUT.parent.mkdir(parents=True, exist_ok=True)
    r = requests.get(API, headers=HEADERS, timeout=30)
    r.raise_for_status()
    rows = r.json()["data"]["table"]["rows"]
    # keep common stocks with a market cap; drop warrants/units (symbols with ^ or ~)
    clean = [(_sanitize(x["symbol"]), _mktcap(x.get("marketCap")))
             for x in rows
             if x.get("symbol") and all(c not in x["symbol"] for c in "^~")]
    clean = [(s, m) for s, m in clean if m > 0]
    clean.sort(key=lambda t: t[1], reverse=True)
    top = clean[:500]
    with open(OUT, "w") as f:
        f.write("Symbol,Series\n")
        for s, _ in top:
            f.write(f"{s},EQ\n")
    print(f"Saved {len(top)} symbols to {OUT} (largest: {top[0][0]})")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
# tools/pull_yfinance_history.py
"""Bulk-pull daily OHLCV from yfinance into the historical_data table.

Writes rows with data_source='yfinance' and plain US tickers (no NSE prefix).
Idempotent per symbol: deletes existing yfinance rows in the window before insert.

Usage:
  PYTHONPATH=. python tools/pull_yfinance_history.py \
      --universe src/data/symbols/nasdaq500.csv \
      --start 2022-05-24 --end 2026-05-24
"""
from __future__ import annotations
import argparse, csv, sys, time
from datetime import datetime, date
from pathlib import Path

import pandas as pd
import yfinance as yf
from sqlalchemy import text

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))
from src.models.database import get_database_manager  # noqa: E402


def load_universe(path: str) -> list[str]:
    syms = []
    with open(path) as f:
        for row in csv.DictReader(f):
            if row.get("Series", "EQ").strip() == "EQ":
                syms.append(row["Symbol"].strip())
    return syms


def insert_symbol(conn, sym: str, df: pd.DataFrame) -> int:
    conn.execute(text("DELETE FROM historical_data WHERE symbol=:s AND data_source='yfinance' "
                      "AND date BETWEEN :a AND :b"),
                 {"s": sym, "a": df.index.min().date(), "b": df.index.max().date()})
    n = 0
    for ts, r in df.iterrows():
        d = ts.date()
        conn.execute(text(
            "INSERT INTO historical_data "
            "(symbol,date,timestamp,open,high,low,close,volume,adj_close,data_source,api_resolution,is_adjusted) "
            "VALUES (:sym,:d,:ts,:o,:h,:l,:c,:v,:ac,'yfinance','1D',false)"),
            {"sym": sym, "d": d, "ts": int(datetime(d.year, d.month, d.day).timestamp()),
             "o": float(r["Open"]), "h": float(r["High"]), "l": float(r["Low"]),
             "c": float(r["Close"]), "v": int(r["Volume"] or 0),
             "ac": float(r.get("Adj Close", r["Close"]))})
        n += 1
    return n


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--universe", required=True)
    ap.add_argument("--start", default="2022-05-24")
    ap.add_argument("--end", default=date.today().isoformat())
    ap.add_argument("--batch", type=int, default=50)
    args = ap.parse_args()

    syms = load_universe(args.universe)
    print(f"Universe: {len(syms)} symbols, {args.start} → {args.end}")
    eng = get_database_manager().engine
    total, failed = 0, []
    for i in range(0, len(syms), args.batch):
        batch = syms[i:i + args.batch]
        data = yf.download(batch, start=args.start, end=args.end, group_by="ticker",
                           auto_adjust=False, threads=True, progress=False)
        with eng.begin() as conn:
            for sym in batch:
                try:
                    sub = data[sym] if len(batch) > 1 else data
                    sub = sub.dropna(subset=["Open", "High", "Low", "Close"])
                    if sub.empty:
                        failed.append(sym); continue
                    total += insert_symbol(conn, sym, sub)
                except Exception as e:  # noqa: BLE001
                    failed.append(sym); print(f"  ! {sym}: {e}", file=sys.stderr)
        print(f"  batch {i//args.batch+1}: {min(i+args.batch,len(syms))}/{len(syms)} done, rows={total}")
        time.sleep(1)
    print(f"DONE rows={total} failed={len(failed)}: {failed[:20]}")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
